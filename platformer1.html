// Start with your first platform createPlatform(0, 0, 0, 10, 1, 10); // Ground platform createPlatform(2, 4, -8, 5, 1, 5); // Small platform createPlatform(-3, 8, -15, 6, 1, 6); // Medium platform createPlatform(1, 12, -20, 6, 1, 6); // Medium platform createPlatform(4, 16, -25, 8, 1, 8); // Large platform createPlatform(-2, 20, -30, 10, 1, 10); // Extra large platform (checkpoint) // Function to generate random numbers between two values function randomInRange(min, max) { return Math.random() * (max - min) + min; } // Create 100 more platforms and obstacles let currentHeight = 0; // Start at ground level for (let i = 1; i <= 100; i++) { let x = randomInRange(i * 2 - 5, i * 2 + 5); // Randomize X position to add variation let y = currentHeight + randomInRange(-3, 3); // Random Y change (up or down) let z = -i * 8 + randomInRange(-5, 5); // Randomize Z to make platforms not all in a straight line // Create platforms with varying sizes and random height adjustments createPlatform(x, y, z, 5 + (i % 5), 1, 5 + (i % 5)); // Add obstacles (spikes) at random positions around the platform let spikeCount = Math.floor(randomInRange(1, 4)); // Random number of spikes per platform (1-3) for (let j = 0; j < spikeCount; j++) { let spikeX = x + randomInRange(-3, 3); // Random horizontal position of spikes let spikeY = y + randomInRange(0, 2); // Random vertical placement (above or on the platform) let spikeZ = z + randomInRange(-3, 3); // Random horizontal displacement for Z axis createObstacle(spikeX, spikeY, spikeZ, 0.5); // Create the spike } // Add a checkpoint every 10 platforms if (i % 10 === 0) { createCheckpoint(x, y + 2, z, 1); // Checkpoint every 10th platform } // Update current height to use for the next platform currentHeight = y; }
